
1. SELECT I.ID, I.SUMMARY, I.DESCRIPTION , IL.NAME AS TASK_STATUS
FROM ISSUES I 
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

2. SELECT I.ID, I. SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

3. SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_OF_TASKS
FROM ISSUES I 
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY U.FIRSTNAME, U.LASTNAME;
